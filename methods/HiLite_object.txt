  //  HiLite_object (text{;object{;text}})  // $1: name of form object to hightlight  // $2: hiLite params object    // $3: name of hiLite pic var  // Written by: Kirk as Designer, Created: 02/16/19, 10:59:30  // ------------------  // Purpose: manage highlighting form object $1  // $2 & $3 are opitonal if hiLite_obj is defined  // the process var 'hiLite_obj' :  //  { picVar: (pic object name), params: {(default hilite object)} }C_TEXT($1;$formObj_t)C_OBJECT($2;$params;hiLite_obj)C_TEXT($3;$picVar_name)C_BOOLEAN($continue)C_TEXT($errMsg)C_LONGINT($x;$y;$x1;$y1;$xPad;$yPad;$fontSize)$formObj_t:=$1$continue:=TrueCase of : (Count parameters<3) & (hiLite_obj=Null)$errMsg:="The hiLite object is not defined and you did not pass enough parameters."$continue:=False: (Count parameters>1)If (hiLite_obj.params=Null)hiLite_obj.params:=$2End if $params:=$2  //  this is the params used this timeIf (Count parameters>2)hiLite_obj.picVar:=$3End if Else   //  use the defined params$params:=hiLite_obj.paramsEnd case   // --------------------------------------------------------If ($continue)$picVar_name:=hiLite_obj.picVarOBJECT GET COORDINATES(*;$formObj_t;$x;$y;$x1;$y1)  // padding  // padding is a function of the font size of the hiLite object$fontSize:=OBJECT Get font size(*;$formObj_t)Case of : (($fontSize)<=12)  // the min pad$xPad:=2$yPad:=1: (($fontSize)<=18)$xPad:=4$yPad:=2Else $xPad:=6$yPad:=3End case   // --------------------------------------------------------$x:=$x-$xPadIf ($params.type="frame") | ($params.type="halfFrame")$x:=$x-$params.strokeEnd if $y:=$y-$yPad$x1:=$x1+$xPad$y1:=$y1+$yPadIf ($params.type="frame") | ($params.type="halfFrame")$y1:=$y1+($params.stroke*0.5)End if OBJECT SET COORDINATES(*;$picVar_name;$x;$y;$x1;$y1)OBJECT SET VISIBLE(*;$picVar_name;True)  // update the width & height in $obj$params.width:=$x1-$x$params.height:=$y1-$yOBJECT Get pointer(Object named;$picVar_name)->:=HiLite_draw_pic ($params)End if If ($errMsg#"")ALERT($errMsg)End if
