  //  HiLite_get_preDefinedParam (text) -> object  // $1: a name for a preDefined param  // $0: the object  // Written by: Kirk as Designer, Created: 02/16/19, 17:33:50  // ------------------  // Purpose: a handy way to refernce some pre-defined paramsC_TEXT($1;$name)C_OBJECT($0;$params)If (Count parameters=1)$name:=$1End if Case of : ($name="halfFrame")$params:=New object(\"type";"halfFrame";\"opacity";40;\"fillColor";"white";\"stroke";3;\"lineColor";"gray"): ($name="redFrame")$params:=New object(\"type";"frame";\"opacity";100;\"fillColor";"white";\"stroke";3;\"lineColor";"red"): ($name="greenFrame")$params:=New object(\"type";"frame";\"opacity";100;\"fillColor";"white";\"stroke";3;\"lineColor";"#97fa4b"): ($name="gradient")$params:=New object(\"type";"grad";\"startColor";"#656d5c:10";\"endColor";"white";\"rot";90)Else   // plain, vanilla white$params:=New object(\"type";"solid";\"opacity";100;\"fillColor";"white")End case $0:=$params
